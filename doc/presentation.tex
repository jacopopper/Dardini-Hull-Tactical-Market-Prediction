\section{Introduction \& Overview}

% ===== Slide 3: Competition Overview =====
\begin{frame}{The Kaggle Challenge}
\begin{itemize}
    \item \textbf{Goal}: predict daily S\&P 500 excess returns and output a weight $w \in [0, 2]$
    \item \textbf{Metric}: adjusted Sharpe ratio, penalizes both excess volatility and return underperformance
    \item \textbf{Data}: \texttt{train.csv} only, $\sim$9k anonymized trading days (macro, sentiment, pricing, volatility features)
\end{itemize}
\vspace{0.3cm}
The adjusted Sharpe metric is what shapes the whole strategy: staying close to the index ($w = 1$) is often better than a confident wrong prediction.
\end{frame}

% ===== Slide 4: The Adjusted Sharpe Metric =====
\begin{frame}{Adjusted Sharpe Ratio}
\begin{align*}
    \text{adjusted\_sharpe} &= \frac{\text{Sharpe}}{\text{vol\_penalty} \times \text{return\_penalty}} \\[0.1cm]
    \text{vol\_penalty} &= 1 + \max\left(0,\; \frac{\sigma_{\text{strat}}}{\sigma_{\text{mkt}}} - 1.2\right) \\[0.05cm]
    \text{return\_penalty} &= 1 + \frac{\text{gap}^2}{100} \\[0.05cm]
    \text{gap} &= \max\bigl(0,\; (\mu_{\text{mkt}} - \mu_{\text{strat}}) \times 100 \times 252\bigr)
\end{align*}
\vspace{0.1cm}
The \textbf{quadratic} return penalty is what makes sustained underperformance really hurt, falling 5\% behind the market annualized multiplies the denominator by $\sim$1.25. This is why I default to $w = 1$ unless the signal is genuinely strong.
\end{frame}

% ===== Slide 5: Data Overview =====
\begin{frame}{Data Description}
\begin{itemize}
    \item 8\,990 trading days: forward returns, risk-free rate, and anonymized features
    \begin{itemize}
        \item I1--I5 (interest rates), E1--E12 (economic), S1--S5 (sentiment), P1--P11 (pricing), V1--V13 (volatility)
    \end{itemize}
    \item Split chronologically into three non-overlapping blocks:
\end{itemize}
\vspace{0.15cm}
\centering
\small
\begin{tabular}{ccc}
    \toprule
    \textbf{Tuning block} & \textbf{Purge gap} & \textbf{Holdout} \\
    \midrule
    7\,290 rows & 200 rows & 1\,500 rows \\
    \bottomrule
\end{tabular}

\vspace{0.15cm}
\raggedright\footnotesize
The 200-row purge gap sits between tuning and holdout to prevent autocorrelation leakage at the boundary.
\end{frame}

% ===== Slide 8: Volatility Proxy =====
\begin{frame}{Volatility Proxy}
The V1--V13 columns are anonymized, but I treat their cross-sectional mean as a VIX-like fear gauge:
\[
\text{VIX Proxy}_t = \frac{1}{13} \sum_{i=1}^{13} V_{i,t}
\]
\vspace{0.1cm}
From this single signal I derive four features fed to both the model and the regime detector:
\begin{enumerate}
    \item \textbf{VIX proxy}, the raw level
    \item \textbf{VIX percentile rank}, where it sits in the training distribution (frozen)
    \item \textbf{VIX term-slope}, V13 $-$ V1, a rough volatility term structure
    \item \textbf{VIX delta}, day-over-day change, momentum of fear
\end{enumerate}
\end{frame}

\section{Market Regime Detection}

% ===== Slide 9: Regime Detection =====
\begin{frame}{Market Regime Detection}
I fit a \textbf{Gaussian HMM} with 3 hidden states and diagonal covariance on four features:
\texttt{vix\_proxy\_pct}, \texttt{roll\_vol\_20}, \texttt{momentum\_20}, \texttt{autocorr\_20}.

\vspace{0.25cm}
Each row gets:
\begin{itemize}
    \item \texttt{regime\_state}, Viterbi-decoded hidden state
    \item \texttt{regime\_prob\_\{0,1,2\}}, posterior probabilities
    \item \texttt{regime\_stability}, confidence in the current assignment
    \item \texttt{regime\_label}, semantic label (\texttt{bull\_low\_vol}, \texttt{bear\_high\_vol}, \texttt{transition}) assigned by comparing training-set mean returns and volatility across states
\end{itemize}
\end{frame}

\section{Model \& Trading Strategy}

% ===== Slide 10: Return Prediction Model =====
\begin{frame}{Return Prediction Model (XGBoost)}
I train three XGBoost models simultaneously on \texttt{market\_forward\_excess\_returns}:
\vspace{0.15cm}
\begin{center}
\small
\begin{tabular}{lcc}
\toprule
\textbf{Model} & \textbf{Objective} & \textbf{Purpose} \\
\midrule
Point model   & \texttt{reg:squarederror}  & Direction \& magnitude \\
Lower quantile (Q10) & \texttt{reg:quantileerror}, $\alpha=0.1$ & Lower bound \\
Upper quantile (Q90) & \texttt{reg:quantileerror}, $\alpha=0.9$ & Upper bound \\
\bottomrule
\end{tabular}
\end{center}
\vspace{0.2cm}
\textbf{Confidence signal}: how large the point prediction is relative to the interval width,
\[
\text{raw\_conf} = \frac{|\text{point\_pred}|}{\max(\text{Q90} - \text{Q10},\; \varepsilon)} \quad \longrightarrow \quad
\text{confidence} = \operatorname{clip}\!\left(\frac{\text{raw\_conf} - p_{10}}{p_{90} - p_{10}},\; 0,\; 1\right)
\]
$p_{10}, p_{90}$ are computed on training data and frozen, mapping confidence into $[0,1]$.
\end{frame}

% ===== Slide 11: Model Hyperparameters =====
\begin{frame}{Model Hyperparameters \& Regularization}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{XGBoost parameters}
\begin{itemize}
    \item \texttt{max\_depth = 4}
    \item \texttt{learning\_rate = 0.05}
    \item \texttt{subsample = 0.8}
    \item \texttt{colsample\_bytree = 0.8}
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Tree counts}
\begin{itemize}
    \item Point model: \textbf{400 trees}
    \item Quantile bounds: \textbf{150 trees} each, shorter ensembles reduce overfitting to noisy tails
\end{itemize}
\end{column}
\end{columns}
\vspace{0.3cm}
\textbf{Purged walk-forward CV}: expanding training windows with a configurable purge gap before each validation fold. No information crosses the boundary through autocorrelated targets.
\end{frame}

% ===== Slide 12: Position Sizing =====
\begin{frame}{Confidence-Gated Position Sizing}
\small
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\textbf{Weight rule:}
\begin{align*}
w &= 1.0 && \text{if conf} < \text{threshold} \\[2pt]
w &= 1.0 + \text{gain} \cdot \hat{r} \cdot \text{vol\_scale} \cdot \text{reg\_mult} && \text{otherwise}
\end{align*}
\vspace{0.05cm}
\textbf{Volatility scaling} (Moreira \& Muir, 2017):
\[
\text{vol\_scale} = \operatorname{clip}\!\left(\frac{\sigma_{\text{target}}}{\sigma_t},\; 0.5,\; 2.0\right)
\]
\end{column}
\begin{column}{0.40\textwidth}
\textbf{Regime multipliers}
\vspace{0.15cm}
\begin{center}
\small
\begin{tabular}{lr}
\toprule
\textbf{Regime} & \textbf{Mult.} \\
\midrule
Bull, low vol  & 1.30 \\
Normal         & 1.00 \\
Transition     & 0.70 \\
Bear, high vol & 0.50 \\
\bottomrule
\end{tabular}
\end{center}
\end{column}
\end{columns}
\vspace{0.15cm}
\normalsize
When confidence is low I stay at $w = 1$; otherwise I scale risk inversely to volatility and reduce exposure in adverse regimes.
\end{frame}

% ===== Slide 13: Runtime Guards =====
\begin{frame}{Runtime Guards}
Two guards run sequentially after the base weight is computed:

\begin{itemize}
    \item \textbf{Volatility servo}: if rolling strategy vol / market vol exceeds the soft cap (1.149) or hard cap (1.20), I scale the active deviation back to 55\% or 20\% respectively, keeping the vol penalty from triggering.
    \item \textbf{Return-gap guard}: if the strategy has fallen behind the market, I force $w \geq 1.0$ to stop the gap from widening and avoid the quadratic return penalty compounding.
\end{itemize}

\vspace{0.2cm}
Both guards are defensive by design, they reduce downside without capping upside on days where the signal is strong and volatility is in range.
\end{frame}

% ===== Slide 15: Hyperparameter Tuning =====
\begin{frame}{Strategy Tuning}
I fix the modeling pipeline first and tune only the four strategy parameters (confidence threshold, gain, target volatility, soft vol cap) using Optuna with 40 trials.

\vspace{0.2cm}
\begin{itemize}
    \item Untuned baseline mean CV adjusted Sharpe: \textbf{0.7246}
    \item After tuning: \textbf{0.8526}
\end{itemize}

\vspace{0.2cm}
Caching the fold predictions is important here: fitting the feature pipeline and XGBoost models takes most of the compute, so running 40 Optuna trials on top of cached predictions is fast and fully reproducible.
\end{frame}

% ===== Slide 17: Leakage Prevention =====
\begin{frame}{Leakage Prevention}
\begin{table}
\footnotesize
\begin{tabular}{p{4.8cm}p{7.2cm}}
\toprule
\textbf{Risk} & \textbf{Mitigation} \\
\midrule
Feature leakage from future & \texttt{fit\_transform} on training only; apply frozen stats \\
Rolling boundary leakage & 400-row history buffer prepended to validation/holdout blocks \\
Temporal lookahead in CV & Purged walk-forward CV; separate 200-day purge before final holdout \\
Target leakage & Lagged targets only (\texttt{.shift(1)}); target excluded from features \\
Tuning lookahead & Strategy parameters tuned on inner temporal holdout \\
XGBoost non-determinism & Enforce \texttt{nthread=1}, \texttt{tree\_method='hist'}, \texttt{seed=42} \\
\bottomrule
\end{tabular}
\end{table}
\vspace{0.2cm}
Any one of these omissions would inflate the backtest. The purge gap and history buffer are the two that are easiest to miss.
\end{frame}

% ===== Slide 18: Reproducibility =====
\begin{frame}{Reproducibility \& Design Notes}
\textbf{Reproducibility}:
\begin{itemize}
    \item All random seeds set to \texttt{42} (XGBoost, sklearn, HMM)
    \item \texttt{nthread=1} and \texttt{tree\_method='hist'} for cross-platform determinism
\end{itemize}
\vspace{0.3cm}
\textbf{Why only 29.5\% active days?}

Most days the Q10--Q90 interval is wide relative to the point estimate, so raw confidence is low and I stay at $w = 1$. Active days tend to cluster around trend-consistent predictions in low-volatility bull periods, exactly when the model's signal-to-noise is highest and the vol penalty is least likely to trigger.
\end{frame}

% ===== Slide 19: Summary =====
\begin{frame}{Summary}
\begin{itemize}
    \item The quadratic return penalty in the adjusted Sharpe is the main design constraint, it forces a conservative default.
    \item Confidence-gating on the quantile interval width is what keeps the strategy from overtrading on noisy predictions.
    \item Volatility scaling (Moreira \& Muir, 2017) and regime multipliers modulate the active bet size without changing the direction.
    \item Purged walk-forward CV with cached predictions makes the tuning stage both fast and leak-free.
\end{itemize}
\end{frame}
